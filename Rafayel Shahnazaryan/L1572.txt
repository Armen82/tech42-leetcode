class Solution {
    public int diagonalSum(int[][] mat) {
        
        int result = 0;
        int j;
        
        for(int i = 0; i < mat.length; i++) {
            if(i != (mat.length-i-1)) {
                result+=mat[i][i];   
                result+=mat[i][mat.length-i-1];
            } else {
                System.out.println("a");
                result+=mat[i][i];   
            }
            
        }
        return result;
    }
}
